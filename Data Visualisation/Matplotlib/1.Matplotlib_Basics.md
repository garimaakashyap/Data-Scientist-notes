# Matplotlib Basics - Complete Guide

## Introduction
Matplotlib is Python's most popular plotting library. It's powerful, flexible, and essential for data visualization.

## Import Matplotlib
```python
import matplotlib.pyplot as plt
import numpy as np
```

## 1. Your First Plot

### Simple Line Plot
```python
import matplotlib.pyplot as plt
import numpy as np

# Create data
x = [1, 2, 3, 4, 5]
y = [2, 4, 6, 8, 10]

# Create plot
plt.plot(x, y)

# Show plot
plt.show()
```

### With NumPy Arrays
```python
# Using NumPy (more common)
x = np.array([1, 2, 3, 4, 5])
y = np.array([2, 4, 6, 8, 10])

plt.plot(x, y)
plt.show()
```

## 2. Basic Plot Customization

### Adding Labels and Title
```python
x = np.linspace(0, 10, 100)
y = np.sin(x)

plt.plot(x, y)
plt.xlabel('X-axis Label')      # X-axis label
plt.ylabel('Y-axis Label')      # Y-axis label
plt.title('My First Plot')      # Plot title
plt.show()
```

### Adding Grid
```python
plt.plot(x, y)
plt.xlabel('X-axis')
plt.ylabel('Y-axis')
plt.title('Plot with Grid')
plt.grid(True)  # Add grid
plt.show()
```

### Changing Line Style and Color
```python
x = np.linspace(0, 10, 100)
y = np.sin(x)

# Different line styles
plt.plot(x, y, linestyle='-', color='blue', label='Solid')
plt.plot(x, y+1, linestyle='--', color='red', label='Dashed')
plt.plot(x, y+2, linestyle=':', color='green', label='Dotted')
plt.plot(x, y+3, linestyle='-.', color='orange', label='Dash-dot')

plt.legend()  # Show legend
plt.show()
```

### Shortcuts for Line Style
```python
# Shortcuts
plt.plot(x, y, '-', label='Solid')      # Same as linestyle='-'
plt.plot(x, y, '--', label='Dashed')    # Same as linestyle='--'
plt.plot(x, y, ':', label='Dotted')     # Same as linestyle=':'
plt.plot(x, y, '-.', label='Dash-dot')  # Same as linestyle='-.'
plt.legend()
plt.show()
```

### Color Options
```python
# Named colors
plt.plot(x, y, color='red')
plt.plot(x, y, color='blue')
plt.plot(x, y, color='green')

# RGB values (0-1)
plt.plot(x, y, color=(0.5, 0.3, 0.8))

# Hex colors
plt.plot(x, y, color='#FF5733')

# Single letter shortcuts
plt.plot(x, y, 'r')  # red
plt.plot(x, y, 'b')  # blue
plt.plot(x, y, 'g')  # green
plt.plot(x, y, 'k')  # black
plt.plot(x, y, 'm')  # magenta
plt.plot(x, y, 'c')  # cyan
```

### Combining Style and Color
```python
# Format: '[color][marker][line]'
plt.plot(x, y, 'r-')    # Red solid line
plt.plot(x, y, 'b--')  # Blue dashed line
plt.plot(x, y, 'g:')   # Green dotted line
plt.plot(x, y, 'ko-')  # Black line with circle markers
```

## 3. Different Plot Types

### Scatter Plot
```python
x = np.random.randn(100)
y = np.random.randn(100)

plt.scatter(x, y)
plt.xlabel('X values')
plt.ylabel('Y values')
plt.title('Scatter Plot')
plt.show()
```

### Customizing Scatter Plot
```python
x = np.random.randn(100)
y = np.random.randn(100)
colors = np.random.rand(100)
sizes = 1000 * np.random.rand(100)

plt.scatter(x, y, c=colors, s=sizes, alpha=0.5)
plt.xlabel('X values')
plt.ylabel('Y values')
plt.title('Customized Scatter Plot')
plt.colorbar()  # Show color scale
plt.show()
```

### Bar Chart
```python
categories = ['A', 'B', 'C', 'D', 'E']
values = [23, 45, 56, 78, 32]

plt.bar(categories, values)
plt.xlabel('Categories')
plt.ylabel('Values')
plt.title('Bar Chart')
plt.show()
```

### Horizontal Bar Chart
```python
plt.barh(categories, values)  # Horizontal bars
plt.xlabel('Values')
plt.ylabel('Categories')
plt.title('Horizontal Bar Chart')
plt.show()
```

### Histogram
```python
data = np.random.normal(100, 15, 1000)

plt.hist(data, bins=30)
plt.xlabel('Value')
plt.ylabel('Frequency')
plt.title('Histogram')
plt.show()
```

### Pie Chart
```python
sizes = [15, 30, 45, 10]
labels = ['A', 'B', 'C', 'D']
colors = ['red', 'blue', 'green', 'orange']

plt.pie(sizes, labels=labels, colors=colors, autopct='%1.1f%%')
plt.title('Pie Chart')
plt.show()
```

## 4. Multiple Plots

### Multiple Lines on Same Plot
```python
x = np.linspace(0, 10, 100)

plt.plot(x, np.sin(x), label='sin(x)')
plt.plot(x, np.cos(x), label='cos(x)')
plt.plot(x, np.sin(x) * np.cos(x), label='sin(x)*cos(x)')

plt.xlabel('X')
plt.ylabel('Y')
plt.title('Multiple Lines')
plt.legend()
plt.grid(True)
plt.show()
```

### Subplots (Multiple Plots)
```python
# Create 2x2 subplots
fig, axes = plt.subplots(2, 2, figsize=(10, 8))

# Plot 1
x = np.linspace(0, 10, 100)
axes[0, 0].plot(x, np.sin(x))
axes[0, 0].set_title('Sin(x)')
axes[0, 0].set_xlabel('X')
axes[0, 0].set_ylabel('Y')

# Plot 2
axes[0, 1].plot(x, np.cos(x), 'r')
axes[0, 1].set_title('Cos(x)')

# Plot 3
axes[1, 0].scatter(np.random.randn(100), np.random.randn(100))
axes[1, 0].set_title('Scatter Plot')

# Plot 4
axes[1, 1].bar(['A', 'B', 'C'], [10, 20, 30])
axes[1, 1].set_title('Bar Chart')

plt.tight_layout()  # Adjust spacing
plt.show()
```

### Subplot Shortcuts
```python
# Using plt.subplot(rows, cols, index)
plt.subplot(2, 2, 1)
plt.plot(x, np.sin(x))
plt.title('Plot 1')

plt.subplot(2, 2, 2)
plt.plot(x, np.cos(x))
plt.title('Plot 2')

plt.subplot(2, 2, 3)
plt.scatter(np.random.randn(50), np.random.randn(50))
plt.title('Plot 3')

plt.subplot(2, 2, 4)
plt.bar(['A', 'B'], [10, 20])
plt.title('Plot 4')

plt.tight_layout()
plt.show()
```

## 5. Figure and Axes Objects

### Understanding Figure and Axes
```python
# Explicit figure and axes
fig, ax = plt.subplots()

x = np.linspace(0, 10, 100)
ax.plot(x, np.sin(x))
ax.set_xlabel('X')
ax.set_ylabel('Y')
ax.set_title('Using Axes Object')
ax.grid(True)

plt.show()
```

### Figure Size
```python
# Set figure size
fig, ax = plt.subplots(figsize=(10, 6))  # width, height in inches

x = np.linspace(0, 10, 100)
ax.plot(x, np.sin(x))
plt.show()
```

### Saving Figures
```python
fig, ax = plt.subplots()
x = np.linspace(0, 10, 100)
ax.plot(x, np.sin(x))
ax.set_title('My Plot')

# Save figure
plt.savefig('my_plot.png')  # PNG format
plt.savefig('my_plot.pdf')   # PDF format
plt.savefig('my_plot.jpg', dpi=300)  # High resolution
plt.show()
```

### Text Annotations
```python

plt.plot(x, y)

plt.ylabel('Y')
```

### Background Color
```python
fig, ax = plt.subplots()

ax.plot(x, np.sin(x))

ax.set_facecolor('lightgray')  # Axes background

```

## 7. Styling

### Using Styles

```python

for style in styles:

    plt.figure()
```


## 10. Complete Example

```python
import matplotlib.pyplot as plt
import numpy as np

# Create data
x = np.linspace(0, 2*np.pi, 100)
y1 = np.sin(x)
y2 = np.cos(x)

# Create figure
fig, ax = plt.subplots(figsize=(10, 6))

# Plot multiple lines
ax.plot(x, y1, 'b-', linewidth=2, label='sin(x)')
ax.plot(x, y2, 'r--', linewidth=2, label='cos(x)')

# Customize
ax.set_xlabel('X (radians)', fontsize=12)
ax.set_ylabel('Y', fontsize=12)
ax.set_title('Sine and Cosine Functions', fontsize=14, fontweight='bold')
ax.grid(True, alpha=0.3)
ax.legend(loc='upper right')

# Set limits
ax.set_xlim(0, 2*np.pi)
ax.set_ylim(-1.5, 1.5)

# Add text
ax.text(np.pi/2, 1, 'Maximum', fontsize=10, 
        bbox=dict(boxstyle='round', facecolor='yellow', alpha=0.5))

plt.tight_layout()
plt.savefig('sine_cosine.png', dpi=300)
plt.show()
```

## Key Takeaways

1. **Import:** `import matplotlib.pyplot as plt`
2. **Basic plot:** `plt.plot(x, y)`
3. **Labels:** `plt.xlabel()`, `plt.ylabel()`, `plt.title()`
4. **Show:** `plt.show()` (always call this!)
5. **Save:** `plt.savefig('filename.png')`
6. **Subplots:** `plt.subplots(rows, cols)`
7. **Styles:** `plt.style.use('style_name')`
8. **Legend:** `plt.legend()` for multiple lines
9. **Grid:** `plt.grid(True)` for grid lines
10. **Figure size:** `figsize=(width, height)`

## Common Mistakes

❌ **Forgetting plt.show()**
```python
plt.plot(x, y)  # Plot created but not displayed!
# Missing: plt.show()
```

✅ **Correct:**
```python
plt.plot(x, y)
plt.show()
```

❌ **Not using NumPy arrays**
```python
x = [1, 2, 3, 4, 5]  # Works but slower
```

✅ **Better:**
```python
x = np.array([1, 2, 3, 4, 5])  # Faster, more features
```

❌ **Not setting figure size for subplots**
```python
fig, ax = plt.subplots(2, 2)  # Might be too small
```

✅ **Better:**
```python
fig, ax = plt.subplots(2, 2, figsize=(10, 8))
```
